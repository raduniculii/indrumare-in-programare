import random

Nr_initial_puncte_jucator1=167
Nr_initial_puncte_jucator2=167
Nr_incercari=0

print("""Hai sa jucam o partida de table! 
Introdu numele primului jucator:""")
Numele_primului_jucator=input()
print ("Introdu numele celui de-al doilea jucator:")
Numele_al_doilea_jucator=input()

zar1_jucator1=int(random.randint(1,6))
zar1_jucator2=int(random.randint(1,6))
print("Zar 1 jucator 1: " + str(zar1_jucator1))
print("Zar 1 jucator 2: " + str(zar1_jucator1))
zar2_jucator1=int(random.randint(1,6))
zar2_jucator2=int(random.randint(1,6))
print("Zar 2 jucator 1: " + str(zar2_jucator1))
print("Zar 2 jucator 2: " + str(zar2_jucator1))
print("Suma zarurilor jucatorului 1: " + str(int(zar1_jucator1) + int(zar2_jucator1)))
print("Suma zarurilor jucatorului 2: " + str(int(zar1_jucator2) + int(zar2_jucator2)))
Nr_puncte_ramase_jucator1=int(Nr_initial_puncte_jucator1-(zar1_jucator1+zar2_jucator1))
Nr_puncte_ramase_jucator2=int(Nr_initial_puncte_jucator2-(zar1_jucator2+zar2_jucator2))
print("Nr puncte ramase jucator 1: " + str(Nr_puncte_ramase_jucator1))
print("Nr puncte ramase jucator 2: " + str(Nr_puncte_ramase_jucator2))

while (Nr_puncte_ramase_jucator1 | Nr_puncte_ramase_jucator2) > 0:
    zar1_jucator1=int(random.randint(1,6))
    zar2_jucator1=int(random.randint(1,6))
    zar1_jucator2=int(random.randint(1,6))
    zar2_jucator2=int(random.randint(1,6))
    print("Zar 1 jucator 1: " + str(zar1_jucator1))
    print("Zar 2 jucator 1: " + str(zar2_jucator1))
    print("Zar 1 jucator 2: " + str(zar1_jucator2))
    print("Zar 2 jucator 2: " + str(zar2_jucator2))
    print("Nr puncte ramase jucator 1: " + str(Nr_puncte_ramase_jucator1))
    print("Nr puncte ramase jucator 2: " + str(Nr_puncte_ramase_jucator2))
    print("Nr incercari curente: " + str(Nr_incercari))
    Nr_puncte_intermediar_jucator1=int(Nr_puncte_ramase_jucator1-(zar1_jucator1+zar2_jucator1))
    Nr_puncte_intermediar_jucator2=int(Nr_puncte_ramase_jucator2-(zar1_jucator2+zar2_jucator2))
    Nr_puncte_ramase_jucator1=Nr_puncte_intermediar_jucator1
    Nr_puncte_ramase_jucator2=Nr_puncte_intermediar_jucator2
    Nr_incercari += 1

if Nr_puncte_ramase_jucator1 < Nr_puncte_ramase_jucator2:
    print(Numele_primului_jucator +" a castigat, felicitari!")
elif Nr_puncte_ramase_jucator1 == Nr_puncte_ramase_jucator2:
    print("Egalitate, felicitari amandurora.")
else:
    print(Numele_al_doilea_jucator + " a castigat, felicitari!")
